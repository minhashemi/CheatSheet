    \begin{RTL}
    اصول احتمال کولموگروف: $\sum P_i = 1\qquad P\in[0,1] \quad$
    
    جمع احتمال: $P(A \cup B) = P(A) + P(B) - P(A \cap B) \quad $
    
    احتمال متمم: $P(\bar{A}) = 1 - P(A) \quad$
    
    استقلال:
    \begin{itemize}
    \item وقوع یکی از پیشامد‌ها روی دیگری تاثیر ندارد
    \item $P(A\cap B) = P(A)P(B)$
    \item $P(A|B)=P(A) \quad or\quad P(B|A)=P(B)$
    \item $P(A|B)=P(A|\bar{B}) \quad or\quad P(B|A)=P(B|\bar{A})$
    \end{itemize}
    
    احتمال شرطی:
    \begin{itemize}
    \item $P(A|B) = \frac{P(A \cap B)}{P(B)}$
    \item $P(A \cap B) = P(B) \times P(A|B) = P(A) \times P(B|A)$
    \end{itemize}
    
    قانون بیز: $P(A_i|B)= \frac{P(B|A_i)P(A_i)}{\sum_{i=1}^{n}P(A_i)P(B|A_i)} \quad$
    \end{RTL}

%------------ Maximum and Minimum of Independent Variables ---------------
\begin{tikzpicture}
\node [mybox] (box){%
    \begin{minipage}{0.46\textwidth}
    
\begin{RTL}
\textit{*برای متغیر تصادفی $X_i$، cdf را با $F_{X_i}(x)$ و pdf را با $f_{X_i}(x)$ نشان می‌دهیم*}

بیشینه ($V=max\{X_1, X_2, ... , X_n\}$)
\begin{align*}
        F_V(v)& = P(V \leq v) = P(X_1 \leq v, X_2 \leq v,..., X_n \leq v)\\
                & = P(X_1 \leq v)P(X_2 \leq v)...P(X_n \leq v) = \prod_{i=1}^n F_{X_i}(v)\\     
        f_V(v)&= F_V'(v) = \frac{d}{dv}F_V(v)\\
\end{align*}

کمینه ($U=min\{X_1, X_2, ... , X_n\}$)
\begin{align*}
        F_U(u)&= P(U \leq u) = 1-P(U>u) = 1-P(X_1>u, X_2>u,...,X_n>u)\\
                &= 1-P(X_1>u)P(X_2>u)...P(X_n>u)\\
                &= 1-\prod_{i=1}^n [1-F_{X_i}(u)]\\
                f_U(u)& =\frac{d}{du} F_U(u) 
\end{align*}
\end{RTL}

    \end{minipage}

    اگر $X_1$, $X_2$,..., $X_n$، $n$ نمونه تصادفی از یک جامعه/توزیع با میانگین $\mu$ و واریانس $\sigma^2$ باشند، برای $n$ به اندازه بزرگ ($n \geq 20$) آنگاه:

    $$T = X_1 + X_2 + ... + X_n = n\overline{X} \qquad T \sim N(n\mu, n\sigma^2)$$
    \end{RTL}

    \begin{RTL}
    \underline{برنولی}
    
    فضای نمونه دو حالته.
    \begin{LTR}
    
        $Pr(Success) = p, \quad Pr(Failure) = 1 - p$
    \end{LTR}
    توزیع برنولی:
    \begin{LTR}
    $X \sim Bernoulli(p)$
    
    pmf: $P(X=x) = p^x(1-p)^{1-x}, \quad x=0,1$
    \end{LTR}
    \end{RTL}
    \begin{tabular}{lp{8cm} l}
    	میانگین: &$\mu = \mathbb{E}[X] = p$ \\
    	واریانس: &$\sigma^2 = Var(X) = p(1-p)$\\
	\end{tabular}\\
	
\begin{RTL}
    \underline{دوجمله‌ای}
    
    متغیر تصادفی دوجمله‌ای $X$، تعداد موفقیت‌ها در $n$ تلاش مستقل که همه موفقیت‌ها، احتمال برابر $p$ دارند.
    
    توزیع دوجمله‌ای:
        \begin{LTR}
        $X \sim Bin(n, p)$
        \renewcommand{\arraystretch}{1.5}

        \begin{tabular}{c l}
        pmf& $P(X=x) = {n \choose x} p^x(1-p)^{n-x}$ \\
        cdf& $P(X\leq x) = \sum_{i=0}^{x} {n \choose i} p^i(1-p)^{n-i}$ 
        \end{tabular}
            \end{LTR}
    
\end{RTL}
    
    \begin{tabular}{lp{8cm} l}
    	میانگین: &$\mu = \mathbb{E}[X] = np$ \\
    	واریانس: &$\sigma^2 = Var(X) = np(1-p)$\\
	\end{tabular}
\\ \\
	\begin{RTL}
	\underline{تخمین نرمال توزیع دوجمله‌ای}
	
	    اگر $X \sim Bin(n,p)$ و با تصحیح پیوستگی، داشته باشیم  $np \geq 5$ و $n(1-p) \geq 5$، آنگاه :
    $$X \sim N(np, np(1-p)), approx.$$
	\end{RTL}

       \begin{itemize}

        \item $H_1$ : مشاهده ما. ادعای ما. چیزی که تلاش می‌کنیم اثبات کنیم.
    \end{itemize}
    
    در آزمون فرض، صرفا با مقایسه دو فرضیه، درستی یا نادرستی $H_0$ را بررسی می‌کنیم.
    
    \underline{انواع خطا}
    \begin{itemize}
    \item
    خطای نوع اول $\alpha$ : رد $H_0$ به شرط درستی $H_0$ یا به عبارتی: 
    $Pr( H_0 \text{\ding{55}} | H_0)$
    
    \item
    خطای نوع دوم $\beta$ : پذیرفتن $H_0$ به شرط نادرستی $H_0$ یا به عبارتی:    
        $Pr( H_0 \checkmark | H_1)$
    \end{itemize}
    
$H_1$ در خطای نوع اول دخیل نیست.

خطای نوع اول، علاوه بر $H_0$ به اندازه نمونه نیز بستگی دارد.
\\ \\
    خطای اول و دوم، رفتار الاکلنگی ($trade-off$) دارند.
    \\ \\
$\alpha$ خطای مهمتری از $\beta$ است. مثال قاضی
\\ \\
با افزایش اندازه نمونه، $\alpha$ کاهش می‌یابد.
\\ \\

\underline{تابع قدرت}:
مقدار $1-\beta$ را قدرت تست و برای فرضیات مرکب، تابع قدرت گوییم.

دوست داریم $H_1$ را بپذیریم. پس دوست داریم $\alpha$ کم و قدرت تست بالا داشته باشیم.
\\ \\
\underline{p-value}:

کمترین احتمال خطای نوع اول که آماره آزمون موجب رد فرض صفر شود، به شرط آنکه فرض صفر صحیح باشد:
$p-value = Pr(X>x | H_0)$

$\alpha > p-value \Rightarrow \text{ reject}H_0$
}

\fbox{
